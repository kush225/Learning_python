#for converting multiple files at once
import os
for f in os.listdir('.'):
    if f.endswith('.jpg'):
        i = Image.open(f)
        f_name, f_ext = os.path.splitext(f)
        i.save('pngs/{}.png'.format(f_name))

#files in pngs folder
print(os.listdir('pngs'))
